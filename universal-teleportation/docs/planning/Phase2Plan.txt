cluster/node_registry.py
class NodeRegistry:
    """
    Keeps track of all nodes capable of receiving teleported workloads.
    """

    def __init__(self):
        self.nodes = {}

    def register_node(self, node_id, address, role="worker"):
        self.nodes[node_id] = {
            "address": address,
            "role": role,
        }

    def get_node(self, node_id):
        return self.nodes.get(node_id)

    def list_nodes(self):
        return self.nodes

cluster/node_manager.py
from node_registry import NodeRegistry

class NodeManager:

    def __init__(self):
        self.registry = NodeRegistry()

    def add_node(self, node_id, address):
        self.registry.register_node(node_id, address)

    def get_node(self, node_id):
        return self.registry.get_node(node_id)

    def list_nodes(self):
        return self.registry.list_nodes()
