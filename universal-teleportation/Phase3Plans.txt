🐳 docker_adapter.py
This adapter connects teleportation with Docker.
import subprocess

class DockerAdapter:
    """
    Adapter for interacting with Docker containers.
    """

    def list_containers(self):
        result = subprocess.run(
            ["docker", "ps", "--format", "{{.ID}}"],
            capture_output=True,
            text=True
        )

        return result.stdout.strip().split("\n")

    def inspect_container(self, container_id):
        result = subprocess.run(
            ["docker", "inspect", container_id],
            capture_output=True,
            text=True
        )

        return result.stdout

📦 container_checkpoint.py
Creates a checkpoint using Docker + CRIU.
import subprocess

def checkpoint_container(container_id, checkpoint_name="uat-checkpoint"):
    """
    Create container checkpoint.
    """

    subprocess.run([
        "docker",
        "checkpoint",
        "create",
        container_id,
        checkpoint_name
    ])

    return checkpoint_name
Checkpoint output:
/var/lib/docker/containers/<id>/checkpoints/

🔄 container_restore.py
Restores a container from checkpoint.
import subprocess

def restore_container(container_id, checkpoint_name):
    """
    Restore container from checkpoint.
    """

    subprocess.run([
        "docker",
        "start",
        "--checkpoint",
        checkpoint_name,
        container_id
    ])
